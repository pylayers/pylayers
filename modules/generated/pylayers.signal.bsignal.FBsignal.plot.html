Examples

>>> from numpy import *
>>> from scipy import *
>>> S = FBsignal()
>>> S.x = arange(100)
>>> S.y = cos(2*pi*S.x)+1j*sin(3*pi*S.x+pi/3)
>>> fig,ax = S.plot()
>>> plt.show()
